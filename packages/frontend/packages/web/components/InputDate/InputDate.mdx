# @britania-crm/web-componens/InputDate
```js
import InputDate from '@britania-crm/web-components/InputDate'
```

This component is a wrapper to [`@britania-crm/web-components/TextField`](/web-components/text-field).

## Live basic usage
```js
import { useState, useCallback } from 'react'

import InputDate from '@britania-crm/web-components/InputDate'
```
<Playground>
  {() => {
    const [value, setValue] = useState('2020-07-19')
    const onChange = useCallback(
      (newDate) => setValue(newDate),
      []
    )
    return (
      <InputDate
        detached
        value={ value }
        onChange={ onChange }
        label="Date"
        name="date"
      />
    )
  }}
</Playground>

## Usage in Form
```js
import flow from 'lodash/fp/flow'
import * as Yup from 'yup'
import { useRef, useCallback } from 'react'
import Form from '@britania-crm/web-components/Form'
import InputDate from '@britania-crm/web-components/InputDate'
import date from '@britania-crm/forms/validators/date.validator'
import minDate from '@britania-crm/forms/validators/minDate.validator'
```
<Playground>
  {() => {
    const formRef = useRef(null)
    const schemaConstructor = useCallback(
      ({ t }) => Yup.object().shape({
        date: flow(
          date({ t }),
          minDate({ t, minDate: '2021-10-05' })
        )(Yup.string())
      }),
      []
    )
    return (
      <Form ref={ formRef } schemaConstructor={ schemaConstructor }>
        <InputDate label="Date" name="date" min="2021-10-05" />
      </Form>
    )
  }}
</Playground>
