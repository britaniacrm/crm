# @britania-crm/web-componens/InputStateRegistration
```js
import InputStateRegistration from '@britania-crm/web-components/InputStateRegistration'
```

This component is a wrapper to [`@britania-crm/web-components/TextField`](/web-components/text-field)..

## Usage in Form
```js
import * as Yup from 'yup'
import { useRef, useCallback } from 'react'
import Form from '@britania-crm/web-components/Form'
import InputStateRegistration from '@britania-crm/web-components/InputStateRegistration'
import stateRegistration from '@britania-crm/forms/validators/stateRegistration.validator'
```
<Playground>
  {() => {
    const formRef = useRef(null)
    const [state, setState] = useState('')
    const schemaConstructor = useCallback(
      ({ t }) => Yup.object().shape({
        state: Yup.string(),
        stateRegistration: stateRegistration({
          t
          // stateFieldName: 'state'
        })(Yup.string())
      }),
      []
    )
    const onChangeState = useCallback(
      (newState) => {
        setState(newState)
        formRef.current.validateField('stateRegistration')
      },
      []
    )
    return (
      <Form ref={ formRef } schemaConstructor={ schemaConstructor }>
        <InputSelect
          name="state"
          label="State"
          onValueChange={ onChangeState }
          options={ [
            { id: 'AC', name: 'AC' },
            { id: 'AM', name: 'AM' },
            { id: 'BA', name: 'BA' },
            { id: 'CE', name: 'CE' },
            { id: 'DF', name: 'DF' },
            { id: 'GO', name: 'GO' },
            { id: 'MT', name: 'MT' },
            { id: 'MG', name: 'MG' },
            { id: 'PA', name: 'PA' },
            { id: 'PB', name: 'PB' },
            { id: 'PR', name: 'PR' },
            { id: 'PE', name: 'PE' },
            { id: 'RJ', name: 'RJ' },

          ] }
        />
        <InputStateRegistration
          label="Inscrição Estadual"
          name="stateRegistration"
          state={ state }
        />
      </Form>
    )
  }}
</Playground>
