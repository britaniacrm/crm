# @britania-crm/web-componens/InputPassword
```js
import Form from '@britania-crm/web-components/Form'
import InputPassword from '@britania-crm/web-components/InputPassword'
```

This component is a wrapper to [`@britania-crm/web-components/TextField`](/web-components/text-field).

## Live basic usage
```js
import { useState, useCallback } from 'react'

import InputPassword from '@britania-crm/web-components/InputPassword'
```
<Playground>
  {() => {
    const [value, setValue] = useState('')
    const onChange = useCallback(
      (event) => setValue(event.target.value),
      []
    )
    return (
      <InputPassword
        detached // important to use it outside form
        value={ value }
        onChange={ onChange }
        label="Password"
        name="password"
      />
    )
  }}
</Playground>

## Usage in Form
```js
import { useRef, useCallback } from 'react'

import * as Yup from 'yup'

import password from '@britania-crm/forms/validators/password'
import Form from '@britania-crm/web-components/Form'
import InputPassword from '@britania-crm/web-components/InputPassword'
```
<Playground>
  {() => {
    const formRef = useRef(null)
    const schemaConstructor = useCallback(
      ({ t }) => Yup.object().shape({ password: password({ t })(Yup.string()) }),
      []
    )
    return (
      <Form ref={ formRef } schemaConstructor={ schemaConstructor }>
        <InputPassword label="Password" name="password" />
      </Form>
    )
  }}
</Playground>
